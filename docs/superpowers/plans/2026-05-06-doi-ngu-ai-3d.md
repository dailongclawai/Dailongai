## Pre-flight assumptions

1. Phase 0 work (sitemap entry, Header.tsx nav, translation keys, public/images/team/*) is **kept as-is**. Only `src/app/doi-ngu-ai/page.tsx` and `styles.css` are replaced.
2. `output: "export"` requires the entire 3D layer to be a client component (`"use client"`). The page.tsx itself stays a server component for `metadata` export.
3. Existing repo 3D pattern is vanilla Three.js (`DnaHelix`, `LuminousFilaments`, `OrganicWaveField`). We deliberately introduce R3F here per Boss's confirmed Q6 — those components are untouched.
4. Static export means `<Canvas>` mounts only on client (effective `ssr: false` automatically because of `"use client"` boundary). Use `next/dynamic` with `ssr: false` for the Scene3D wrapper as a belt-and-suspenders.
5. Header.tsx is fixed positioning. Canvas + scroll content sits behind via `z-index`. Header text remains readable on dark Canvas (existing Header has backdrop blur).

```tsx
"use client";

import { Canvas } from "@react-three/fiber";
import { ScrollControls, useScroll, Html } from "@react-three/drei";
import { useFrame } from "@react-three/fiber";
import { useRef, useState } from "react";

const TOTAL_PAGES = 5; // 5 viewport heights of scroll: hero+4depts/infra/cta

function ScrollDebugHud() {
  const scroll = useScroll();
  const [pct, setPct] = useState(0);
  useFrame(() => {
    const next = Math.round(scroll.offset * 1000) / 10;
    if (next !== pct) setPct(next);
  });
  return (
    <Html
      position={[0, 0, -2]}
      center
      style={{
        color: "#00f2ff",
        fontFamily: "Share Tech Mono, monospace",
        fontSize: 14,
        letterSpacing: "0.2em",
        pointerEvents: "none",
      }}
    >
      SCROLL :: {pct.toFixed(1)}%
    </Html>
  );
}

export default function Scene3D() {
  return (
    <div className="scene3d-wrap">
      <Canvas

        gl={{ antialias: true, alpha: false }}

        <color attach="background" args={["#050810"]} />

        <ScrollControls pages={TOTAL_PAGES} damping={0.18}>
```

```tsx
// src/app/doi-ngu-ai/components/CameraRig.tsx
"use client";

import { useFrame, useThree } from "@react-three/fiber";
import { useScroll } from "@react-three/drei";
import { useMemo } from "react";
import * as THREE from "three";

// 5 waypoints: hero open · nexus · dept1-2 · dept3-4 · infra · cta
const WAYPOINTS: [number, number, number][] = [
  [0, 0.5, 8],     // 0%   hero open — pulled back
  [0, 0, 4.5],     // 18%  nexus established
  [-3, -1.5, 5],   // 40%  dept zone left
  [3, -3.5, 5],    // 60%  dept zone right
  [0, -6, 7.5],    // 80%  infra ring tower (looking up at it)
  [0, -8, 3],      // 100% cta portal entry
];

const LOOKATS: [number, number, number][] = [
  [0, 0, 0],
  [0, 0, 0],
  [-3, -1.5, 0],
  [3, -3.5, 0],
  [0, -6, 0],
  [0, -8, -2],
];

export default function CameraRig() {
  const { camera } = useThree();
  const scroll = useScroll();

  const posCurve = useMemo(
    () =>
      new THREE.CatmullRomCurve3(
        WAYPOINTS.map((p) => new THREE.Vector3(...p)),
        false,
        "catmullrom",
        0.4,
      ),
    [],
  );
  const lookCurve = useMemo(
    () =>
      new THREE.CatmullRomCurve3(
        LOOKATS.map((p) => new THREE.Vector3(...p)),
        false,
        "catmullrom",
        0.4,
      ),
    [],
  );

  useFrame(() => {
    const t = THREE.MathUtils.clamp(scroll.offset, 0, 1);
    const pos = posCurve.getPoint(t);
    const look = lookCurve.getPoint(t);
    camera.position.copy(pos);
    camera.lookAt(look);
  });

  return null;
}
```

```tsx
import { Billboard, Image as DreiImage, useTexture } from "@react-three/drei";
import { useRef } from "react";
import * as THREE from "three";
import { useFrame } from "@react-three/fiber";

type Props = {
  src: string;
  position: [number, number, number];
  size?: number;
  haloColor?: string;
  haloIntensity?: number;
};

export default function AvatarBillboard({
  src,
  position,
  size = 0.7,
  haloColor = "#00f2ff",
  haloIntensity = 1.2,
}: Props) {
  const haloRef = useRef<THREE.Mesh>(null);

  useFrame(({ clock }) => {
    if (haloRef.current) {
      const t = clock.elapsedTime;
      (haloRef.current.material as THREE.MeshBasicMaterial).opacity =
        0.3 + 0.15 * Math.sin(t * 1.4 + position[0]);

  });

  return (
    <Billboard position={position} follow={true}>
      <mesh ref={haloRef} scale={size * 1.6}>
        <circleGeometry args={[1, 64]} />
        <meshBasicMaterial
          color={haloColor}
          transparent
          opacity={0.3}
          blending={THREE.AdditiveBlending}
          depthWrite={false}
        />
      </mesh>
      <DreiImage
        url={src}
        scale={[size, size, 1]}
        radius={size * 0.5}
        toneMapped={false}
      />
    </Billboard>
  );
}
```

```ts
// src/app/doi-ngu-ai/shaders/neuralMesh.glsl.ts
export const vertexShader = /* glsl */ `
varying vec3 vNormal;
varying vec3 vPosition;
uniform float uTime;

// 3D simplex-ish noise (cheap)
float hash(vec3 p) { p = fract(p * 0.3183099 + 0.1); p *= 17.0; return fract(p.x * p.y * p.z * (p.x + p.y + p.z)); }

void main() {
  vec3 displaced = position;
  float n = hash(position * 1.5 + uTime * 0.15);
  displaced += normal * (n - 0.5) * 0.18;

  vNormal = normalize(normalMatrix * normal);
  vPosition = displaced;

  gl_Position = projectionMatrix * modelViewMatrix * vec4(displaced, 1.0);
}
`;

export const fragmentShader = /* glsl */ `
varying vec3 vNormal;
varying vec3 vPosition;
uniform float uTime;
uniform vec3 uColorA;
uniform vec3 uColorB;

void main() {
  float fresnel = pow(1.0 - max(dot(vNormal, vec3(0.0, 0.0, 1.0)), 0.0), 2.5);
  vec3 grad = mix(uColorA, uColorB, 0.5 + 0.5 * sin(vPosition.y * 1.3 + uTime * 0.6));
  vec3 color = grad * (0.4 + fresnel * 1.6);
  // pulse
  float pulse = 0.85 + 0.15 * sin(uTime * 1.8);
  gl_FragColor = vec4(color * pulse, 1.0);
}
`;
```

```tsx
// src/app/doi-ngu-ai/components/NexusCore.tsx
"use client";

import { useMemo, useRef } from "react";
import { useFrame } from "@react-three/fiber";
import * as THREE from "three";
import { vertexShader, fragmentShader } from "../shaders/neuralMesh.glsl";

type Props = { position?: [number, number, number]; scale?: number };

export default function NexusCore({ position = [0, 0, 0], scale = 1.4 }: Props) {
  const matRef = useRef<THREE.ShaderMaterial>(null);

  const uniforms = useMemo(
    () => ({
      uTime: { value: 0 },
      uColorA: { value: new THREE.Color("#00f2ff") },
      uColorB: { value: new THREE.Color("#ff9069") },
    }),
    [],
  );

  useFrame((_, dt) => {
    if (matRef.current) {
      (matRef.current.uniforms.uTime.value as number) += dt;
    }
  });

  return (
    <mesh position={position} scale={scale}>
      <icosahedronGeometry args={[1, 5]} />
      <shaderMaterial
        ref={matRef}
        vertexShader={vertexShader}
        fragmentShader={fragmentShader}
        uniforms={uniforms}
      />
    </mesh>
  );
}
```

```tsx
// src/app/doi-ngu-ai/components/SpatialPanel.tsx
"use client";

import { MeshTransmissionMaterial, Text, RoundedBox } from "@react-three/drei";
import { useFrame } from "@react-three/fiber";
import { useRef } from "react";
import * as THREE from "three";

type Props = {
  position: [number, number, number];
  rotation?: [number, number, number];
  width?: number;
  height?: number;
  title: string;
  subtitle: string;
  accent: string;
  scrollProgress: number; // 0..1, drives shrink-to-center
  centerTarget: [number, number, number];
};

export default function SpatialPanel({
  position,
  rotation = [0, 0, 0],
  width = 1.4,
  height = 0.9,
  title,
  subtitle,
  accent,
  scrollProgress,
  centerTarget,
}: Props) {
  const groupRef = useRef<THREE.Group>(null);

  useFrame(({ clock }) => {
    if (!groupRef.current) return;
    const t = clock.elapsedTime;
    // gentle Y-axis bob + rotate
    groupRef.current.rotation.y = rotation[1] + Math.sin(t * 0.4 + position[0]) * 0.06;

    // shrink + drift to center as scrollProgress 0→1
    const k = THREE.MathUtils.smoothstep(scrollProgress, 0.0, 0.25);
    const scale = 1 - k * 0.95;
    groupRef.current.scale.setScalar(scale);
    groupRef.current.position.set(
      THREE.MathUtils.lerp(position[0], centerTarget[0], k),
      THREE.MathUtils.lerp(position[1], centerTarget[1], k),
      THREE.MathUtils.lerp(position[2], centerTarget[2], k),
    );
  });

  return (
    <group ref={groupRef} position={position} rotation={rotation}>
      <RoundedBox args={[width, height, 0.04]} radius={0.05} smoothness={4}>
        <MeshTransmissionMaterial
          backside
          thickness={0.2}
          roughness={0.05}
          transmission={1}
          ior={1.4}
          chromaticAberration={0.02}
          color="#ffffff"
          attenuationColor={accent}
          attenuationDistance={0.6}
        />
      </RoundedBox>
      <Text
        position={[0, 0.18, 0.04]}
        fontSize={0.12}
        color={accent}
        anchorX="center"
        anchorY="middle"
        font="/fonts/Orbitron-Black.ttf" /* fallback default if missing */
      >
        {title}
      </Text>
      <Text
        position={[0, -0.05, 0.04]}
        fontSize={0.06}
        color="#e8eaed"
        anchorX="center"
        anchorY="middle"
        maxWidth={width * 0.85}
      >
        {subtitle}
      </Text>
    </group>
  );
}
```

```tsx
"use client";

import { useScroll, Html, Text } from "@react-three/drei";
import { useFrame } from "@react-three/fiber";
import { useRef, useMemo } from "react";
import * as THREE from "three";
import AvatarBillboard from "../components/AvatarBillboard";
import { departments, type Department, type Sen } from "../data/agents";

type ZoneConfig = {
  dept: Department;
  center: [number, number, number];
  formation: "2x2" | "triangle" | "2x3";
  scrollRange: [number, number]; // when this zone is "focused"
};

const ZONES: ZoneConfig[] = [
  { dept: departments[0], center: [-3, -1.5, 0], formation: "2x2", scrollRange: [0.35, 0.48] },
  { dept: departments[1], center: [3, -3.5, 0], formation: "2x2", scrollRange: [0.48, 0.6] },
  { dept: departments[2], center: [-3, -5.5, 0], formation: "triangle", scrollRange: [0.6, 0.72] },
  { dept: departments[3], center: [3, -7.5, 0], formation: "2x3", scrollRange: [0.72, 0.85] },
];

function formationOffsets(formation: ZoneConfig["formation"]): [number, number, number][] {
  switch (formation) {
    case "2x2":
      return [
        [-0.7, 0.45, 0], [0.7, 0.45, 0],
        [-0.7, -0.45, 0], [0.7, -0.45, 0],
      ];
    case "triangle":
      return [
        [0, 0.6, 0],
        [-0.7, -0.4, 0], [0.7, -0.4, 0],
      ];
    case "2x3":
      return [
        [-0.9, 0.55, 0], [0, 0.55, 0], [0.9, 0.55, 0],
        [-0.9, -0.55, 0], [0, -0.55, 0], [0.9, -0.55, 0],
      ];
  }
}

function DeptZone({ config }: { config: ZoneConfig }) {
  const scroll = useScroll();
  const groupRef = useRef<THREE.Group>(null);
  const offsets = useMemo(() => formationOffsets(config.formation), [config.formation]);

  useFrame(() => {
    if (!groupRef.current) return;
    const [start, end] = config.scrollRange;
    const t = THREE.MathUtils.clamp((scroll.offset - start) / (end - start), 0, 1);
    // fade in opacity via material flag (drei <Image> respects opacity prop on parent? we use group visibility threshold)
    groupRef.current.visible = scroll.offset > start - 0.05 && scroll.offset < end + 0.1;
    // gentle bob
    groupRef.current.position.y = config.center[1] + Math.sin(scroll.offset * 4 + config.center[0]) * 0.05;
  });

  return (
    <group ref={groupRef} position={config.center}>
      {/* dept header */}
      <Text
        position={[0, 1.2, 0]}
        fontSize={0.22}
        color={config.dept.hex}
        anchorX="center"
        anchorY="middle"
      >
        {`┌──[ ${config.dept.emoji} ${config.dept.label.toUpperCase()} ]──`}
      </Text>

      {/* avatars in formation */}
      {config.dept.members.map((sen, i) => {
        const off = offsets[i];
        return (
          <group key={sen.id} position={off}>
            <AvatarBillboard
              src={`/images/team/${sen.avatar}`}
              position={[0, 0, 0]}
              size={0.45}
              haloColor={config.dept.hex}
            />
            <Html
              position={[0, -0.45, 0]}
              center
              distanceFactor={5}
              style={{
                pointerEvents: "none",
                fontFamily: "Share Tech Mono, monospace",
                color: "#e8eaed",
                fontSize: 11,
                textAlign: "center",
                whiteSpace: "nowrap",
              }}
            >
              <div style={{ color: "#fff", fontFamily: "Orbitron, sans-serif", fontSize: 12, letterSpacing: "0.05em" }}>
                {sen.name.toUpperCase()}
              </div>
              <div style={{ color: config.dept.hex, marginTop: 2 }}>{sen.role}</div>
            </Html>
          </group>
        );
      })}

      {/* dept fog tint via colored point light */}
      <pointLight color={config.dept.hex} intensity={0.6} distance={6} />
    </group>
  );
}

export default function DepartmentZones() {
  return (
    <>
      {ZONES.map((z) => <DeptZone key={z.dept.key} config={z} />)}
    </>
  );
}
```
